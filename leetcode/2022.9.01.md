### [最大数](https://leetcode.cn/problems/largest-number/)

```js
var largestNumber = function(nums) {
    /*
    	通过排序使数组按组合后更大的方向排序
    */
    nums = nums.sort((a, b) => {
        const s1 = a.toString() + b.toString()
        const s2 = b.toString() + a.toString() 
        return Number(s2) - Number(s1)
    })
    return nums.join('')[0] === '0' ? '0' : nums.join('')
};
```
