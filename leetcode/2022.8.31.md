### [验证IP地址](https://leetcode.cn/problems/validate-ip-address/)

```js
var validIPAddress = function(queryIP) {
    function IPv4(s){
        const IPv4List = s.split('.')
        if(IPv4List.length !== 4){
            return 'Neither'
        }else{
            const isIPv4 = IPv4List.every((item) => {
                const num = Number(item)
                return  item.length !== 0 && 
                        (item[0] !== '0' || item.length === 1) && 
                        0 <= num && num <= 255 && 
                        item.split('').every((item1)=> {
                            return '0' <= item1 && item1 <= '9'
                        })
            })
            if(isIPv4){
                return 'IPv4'
            }else{
                return 'Neither'
            }
        }
    }
    function IPv6(s){
        const Ipv6List = s.split(':')
        if(Ipv6List.length !== 8){
            return 'Neither'
        }else{
            const isIPv6 = Ipv6List.every((item) => {
                const len = item.length
                if(1 > len || len > 4){
                    return false
                }else{
                    return item.split('').every((item1) => {
                        return  ('0' <= item1 && item1 <= '9') || 
                                ('a' <= item1 && item1 <= 'f') || 
                                ('A' <= item1 && item1 <= 'F') 
                    })       
                }
            })
            if(isIPv6){
                return 'IPv6'
            }else{
                return 'Neither'
            }
        }
    }
    if(queryIP.indexOf('.') !== -1){
        return IPv4(queryIP)
    }else if(queryIP.indexOf(':') !== -1){
        return IPv6(queryIP)
    }else{
        return 'Neither'
    }
};
```

### [二叉搜索树的第k大节点](https://leetcode.cn/problems/er-cha-sou-suo-shu-de-di-kda-jie-dian-lcof/)

```js
var kthLargest = function(root, k) {
    const stack = []
    const res = []
    let cur = root
    while(cur || stack.length){
        while(cur){
            stack.push(cur)
            cur = cur.left
        }
        cur = stack.pop()
        res.push(cur.val)
        cur = cur.right
    }
    return res.reverse()[k - 1]
};
```

### [ 子集](https://leetcode.cn/problems/subsets/)

```js
var subsets = function(nums) {
    const res = []
    const len = nums.length
    function dfs(i, cur){
        res.push(cur)
        for(let j = i; j < len; j ++){
            dfs(j + 1, cur.concat(nums[j]))
        }
    }
    dfs(0, [])
    return res
};
```

### [路径总和 II](https://leetcode.cn/problems/path-sum-ii/)

难度中等832收藏分享切换为英文接收动态反馈

```js
var pathSum = function(root, targetSum) {
    if(!root){
        return []
    }
    const res = []
    function dfs(node, sum, cur){
        if(sum + node.val === targetSum && !node.left && !node.right){
            return res.push(cur.concat(node.val))
        }
        node.left && dfs(node.left, sum + node.val, cur.concat(node.val))
        node.right && dfs(node.right, sum + node.val, cur.concat(node.val))
    }
    dfs(root, 0, [])
    return res
};
```

### [寻找旋转排序数组中的最小值](https://leetcode.cn/problems/find-minimum-in-rotated-sorted-array/)

```js
var findMin = function(nums) {
    let i = 0
    let j = nums.length - 1
    while(i < j){
        const mid = i + Math.floor((j - i) / 2)
        if(nums[mid] < nums[j]){
            j = mid
        }else{
            i = mid + 1
        }
    } 
    return nums[i]
};
```
