### 多数元素

var majorityElement = function(nums) {
    const len = nums.length
    const map = new Map()
    for(let i = 0; i < len; i ++){
        if(!map.has(nums[i])){
            map.set(nums[i], 1)
        }else{
            map.set(nums[i], map.get(nums[i]) + 1)
        }
    }
    for(const [key, value] of map){
        if(value > len / 2){
            return key
        }
    }
};

### 基本计算器

```js
var calculate = function(s) {
    s = s.trim()
    const len = s.length
    let num = 0
    const stack = []
    let preSign = '+'
    for(let i = 0; i < len; i ++){
        if(!isNaN(Number(s[i])) && s[i] !== ' '){
            num = num * 10 + Number(s[i])
        }
        if(isNaN(Number(s[i])) || i === len - 1){
            switch(preSign){
                case '+':
                    stack.push(num)
                    break
                case '-':
                    stack.push(-1 * num)
                    break
                case '*':
                    stack.push(stack.pop() * num)
                    break
                case '/':
                    stack.push(stack.pop() / num | 0)
                    break
            }
            preSign = s[i]
            num = 0
        }
    }
    let ans = 0
    while(stack.length){
        ans += stack.pop()
    }
    return ans
};
```

### 顺时针打印矩阵

```js
var spiralOrder = function(matrix) {
    if(!matrix.length){
        return []
    }
    const dx = [1, 0, -1, 0]
    const dy = [0, 1, 0, -1]
    const c = matrix.length
    const r = matrix[0].length
    let t = c * r
    const res = []
    let a = 0, b = 0, j = 0, x, y
    while(t --){
        res.push(matrix[a][b])
        matrix[a][b] = 101
        x = a
        y = b
        a += dy[j]
        b += dx[j]
        if(-1 < a && a < c && -1 < b && b < r && matrix[a][b] !== 101){}
        else{
            j = (j + 1) % 4
            a = x + dy[j]
            b = y + dx[j]
        }
    }
    return res
};
```

### 对角线遍历

```js
var findDiagonalOrder = function(mat) {
    const c = mat.length
    const r = mat[0].length
    const res = []
    let t = c * r
    let a = 0, b = 0, flag = true, x, y 
    while(t --){
        res.push(mat[a][b])
        x = a
        y = b
        if(flag){
            a -= 1
            b += 1
        }else{
            a += 1
            b -= 1
        }
        if(-1 >= a || a >= c  || -1 >= b || b >= r){
            if(flag){
                a = x
                b = y + 1
                if(-1 >= a || a >= c  || -1 >= b || b >= r){
                    a = x + 1
                    b = y
                }
            }else{
                a = x + 1
                b = y
                if(-1 >= a || a >= c  || -1 >= b || b >= r){
                    a = x
                    b = y + 1
                }
            }
            flag = !flag
        }
    }
    return res
};
```
