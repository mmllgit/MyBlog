```js
function decounce(fn, delay){
    let timer = null
    return function(...args){
        if(timer) clearTimeout(timer)
        const ctx = this

            timer = null
```

### 手写节流

```js
function throttle(fn, delay){
    let timer = null
    let firstTime = true
    return function(...args){
        const ctx = this
        if(firstTime){
            fn.apply(ctx, args)
            return firstTime = false
        }
        if(timer) return 
        timer = setTimeout(function(){
            clearTimeout(timer)
            timer = null
            fn.apply(ctx, args)
        }, delay)
    }
}
```

### 手写Promise.all

```js
Promise.myAll = function (promises) {
  return new Promise((resolve, reject) => {
    const arr = [];
    let count = 0;
    const len = promises.length;
    for (let i = 0; i < len; i++) {
      promises[i].then(
        (v) => {
          arr[i] = v;
          count++;
          if (count === len) {
            resolve(arr);
          }
        },
        (r) => {
          reject(r);
        }
      );
    }
  });
};
```
